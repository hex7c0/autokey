/**
 * export
 * 
 * @exports autokey
 * @function autokey
 * @params {String} password - user password
 * @params {Boolean} [lodash] - flag
 * @return {AUTOKEY}
 */
module.exports = function autokey(password, lodash) {

    if (lodash) {
        return require(min + 'lodash/index.js')(password);
    }
    return require(min + 'normal/index.js')(password);
};
