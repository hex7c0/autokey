/**

 * @exports autokey
 * @function autokey
 * @params {String} key - user key
 * @params {Boolean} [lodash] - flag
 * @return {AUTOKEY}
 */
module.exports = function autokey(key, lodash) {

    if (lodash) {
        return require(min + 'lodash.js')(key);
    }
    return require(min + 'normal.js')(key);
};
